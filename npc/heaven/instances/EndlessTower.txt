e_tower,81,105,0	script	塔之守護石	406,{

	set .@party_id,getcharid(1);
	set .@md_name$,"無盡之塔";

	set .@etower_timer,checkquest(60200,PLAYTIME); // 1 week
	set .@etower_timer2,checkquest(60201,PLAYTIME); // 4 hours

	switch(.@etower_timer) {
	case -1:
		if (getcharid(0) == getpartyleader(.@party_id,2)) {
			mes "你的隊伍符合進入副本的標準, 現在就預約進入「無盡之塔」嗎?";
			next;
			switch(select("預約創建 "+.@md_name$+":進入副本:返回原來的地方:結束對話")) {
			case 1:
				if (instance_create(.@md_name$) < 0) {
					mes "隊伍名稱: "+getpartyname(.@party_id);
					mes "隊長名稱: "+strcharinfo(0);
					mes "^0000ff"+.@md_name$+" ^000000- 預約失敗!";
					close;
				}
				mes "^0000ff"+.@md_name$+"^000000 - 正在嘗試預約";
				mes "在開放進場之後與「塔之守護石」對話, 選擇「進入副本」即可..";
				close;
			case 2:
				callsub L_Enter,0,1;
			case 3:
				mes "我將把你傳送回原本的地方.";
				close2;
				warp "fayon", 70, 156;
				end;
			case 4:
				close;
			}
		}
		switch(select("進入 "+.@md_name$+":返回原來的地方:結束對話")) {
		case 1:
			callsub L_Enter,1,1;
		case 2:
			mes "我將把你傳送回原本的地方.";
			close2;
			warp "fayon", 70, 156;
			end;
		case 3:
			end;
		}
	case 0:
	case 1:
		if (.@etower_timer2 < 2 && getcharid(1) == etower_partyid) {
			mes "若你的副本已經創建完畢且開放進場, 那麼你就可以進入無盡之塔.";
			next;
			switch(select("進入 "+.@md_name$+":返回原來的地方:結束對話")) {
			case 1:
				callsub L_Enter,0,0;
			case 2:
				mes "我將把你傳送回原本的地方.";
				close2;
				warp "fayon", 70, 156;
				end;
			case 3:
				close;
			}
		} else {
			set .@dun_lim_time, etower_timer + 1800; // 30 minutes
			set .@dun_cur_time,gettimetick(2);
			set .@dun_ent_t,(.@dun_lim_time - .@dun_cur_time);
			set .@dun_h,(.@dun_ent_t / 3600);
			set .@dun_m,(.@dun_ent_t - (.@dun_h * 3600)) / 60;
			set .@dun_s,.@dun_ent_t - ((.@dun_h * 3600) + (.@dun_m * 60));

			mes "由於魔力波動的影響現在無法進入副本, 請等 " + .@dun_h + "時" + .@dun_m + "分" + .@dun_s + "秒 後才能重新進入.";
			close;
		}
	case 2:
		set etower_timer,0;
		if (checkquest(60200) >= 0) erasequest 60200;
		if (checkquest(60201) >= 0) erasequest 60201;
		mes "^0000ff無盡之塔的魔力波動已被消除, 你可以再次生成並進入無盡之塔了.^000000";
		close;
	}

L_Enter:
	switch(instance_enter("無盡之塔")) {
	case IE_OTHER:
		mes "發生不明錯誤.";
		mes "可能是副本名稱設定錯誤, 請聯繫管理員.";
		close;
	case IE_NOINSTANCE:
		mes "無盡之塔副本不存在.";
		mes "隊長尚未創建副本, 你無法進入.";
		close;
	case IE_NOMEMBER:
		mes "加入或組建一個隊伍後, 就能進入副本.";
		close;
	case IE_OK:
		mapannounce "e_tower", getpartyname(getcharid(1))+" 隊伍中的 "+ strcharinfo(0) +" 已經進入無盡之塔副本.",bc_map,"0x00ff99",FW_NORMAL,12;
		if (getarg(1)) {
			set etower_timer,gettimetick(2);
			set etower_partyid, getcharid(1);
			setquest 60200;
			setquest 60201;
		}
		//warp "1@tower",52,354;
		if (getarg(0) == 0) close;
		else end;
	}
}
